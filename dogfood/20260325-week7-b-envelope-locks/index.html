run_and_record() {
  local step="$1"
  local stdout_path="$2"
  local stderr_path="$3"
  local optional="$4"
  shift 4

  local exit_code=0
  if "$@" >"$stdout_path" 2>"$stderr_path"; then
    exit_code=0
  else
    exit_code=$?
  fi

  local status="ok"
  if [[ $exit_code -ne 0 ]]; then
    if [[ "$optional" == "optional" ]]; then
      status="optional-failed"
    else
      status="failed"
    fi
  fi

  printf '%s	%s	%s	%s
' "$step" "$*" "$exit_code" "$status" >> "$STATUS"
}
